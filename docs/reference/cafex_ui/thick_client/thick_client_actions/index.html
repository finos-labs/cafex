        """
        Description:
            |  This method allows user to exit from the application based on the parameters

                            bln_flag = True

                if (bln_flag == False):
                    self.logger.info("Incorrect process name provided")

                                                                           pbool_trim_log=True,
                                                                           pbool_log_local_exceptions=True,

                        pobject_dropdown.select(pstr_value)

    def validate_element(self, pobj_element, **kwargs):

        :param **kwargs: pstr_auto_id,pstr_title,pstr_class_name,pstr_parital_text,pstr_control_type

            pstr_auto_id = kwargs.get("pstr_auto_id", None)
            pstr_title = kwargs.get("pstr_title", None)
            pstr_class_name = kwargs.get("pstr_class_name", None)

            pstr_control_type = kwargs.get("pstr_control_type", None)

            str_class_name = pobj_element.element_info.class_name
            str_control_type = pobj_element.element_info.control_type

            if (pstr_auto_id is not None and pstr_auto_id != ""):
                if (pstr_auto_id.lower().strip() == str_auto_id.lower().strip()):
                    return True
            elif (pstr_title is not None and pstr_title != ""):

                    return True

                if (pstr_class_name.lower().strip() == str_class_name.lower().strip()):
                    return True
            elif (pstr_control_type is not None and str_control_type != ""):
                if (pstr_control_type.lower().strip() == str_control_type.lower().strip()):
                    return True
            elif (pstr_parital_text is not None and str_title != ""):
                if (pstr_parital_text.lower().strip() in str_title.lower().strip()):
                    return True

            else:
                return False
        except Exception as e:

        """
        Description:
        	|  This method is used for basic scroll functionality where the user provides the element/section where scroll bars exist,
        	|  along with scroll count and direction. And on the basis of provided parameters, the scroll will perform
        	|  For custom controls where user can face issues in unique identification of element/section, he/she can use parent_element parameter in this case.

        :param pstr_scroll_direction: This parameter will be used to pass the scroll direction e.g. up, down, left or right
        :type pstr_scroll_direction: String
        :param pstr_scroll_count: This parameter will be used to pass the count of scroll, means how many times user wants to scroll down/up/left/right
        :type pstr_scroll_count: Interger
        :param parent_element: This parameter will be used to pass the object of parent element. It will be used when unique identification of element will be difficult
        :type parent_element: Object
        :param scrollable_element: This parameter will be used to pass the element e.g. specific section where scroll bar applies
        :type scrollable_element: Object

        :return: None
        Examples:
	        |  obj.scrolling(scrollable_element=elemnt,pstr_scroll_count=10,pstr_scroll_direction='down')
	        |  obj.scrolling(scrollable_element=scrollable_element, parent_element=parent_elemnt,pstr_scroll_count=5,pstr_scroll_direction='up')

        .. note::
        	|  For custom controls, this method will work when only 1 scroll bar exists. This is not applicable on section/window where multiple scroll bar exists
        	|  For windows/non-custom Controls: This method will work in both single and multiple scroll bars window/sections

        """
        try:
            self.pstr_scroll_direction = kwargs.get("pstr_scroll_direction", None)
            self.pstr_scroll_count = kwargs.get("pstr_scroll_count", None)
            self.pint_retry_count = kwargs.get("pint_retry_count", None)
            self.parent_element = kwargs.get("parent_element", None)
            self.scrollable_element = kwargs.get("scrollable_element", None)
            scrollable_element_wrapper = self.scrollable_element.wrapper_object()
            if scrollable_element_wrapper.element_info.control_type.lower() != 'custom':
                if (item in self.pstr_scroll_direction.lower() for item in ['up', 'down', 'left', 'right']):
                    if (items in self.pstr_scroll_count.lower() for items in ['page', 'end']):
                        self.scrollable_element.scroll(self.pstr_scroll_direction, self.pstr_scroll_count,
                                                       count=self.pint_retry_count)
                    else:
                        raise Exception('Wrong arguments: amount can be any of "page", "end"')
                else:
                    raise Exception('Wrong arguments: direction can be any of "up", "down", "left", "right"')
            elif (scrollable_element_wrapper.element_info.control_type.lower() == 'custom'):

                                                      pstr_auto_id='PART_LineUpButton')
                                self.mouse_operations(up, 'click', pstr_click_type='left')
                                self.pint_temp_count += 1
                        elif (self.pstr_scroll_direction.lower() == 'down'):
                            while (self.pint_temp_count < self.pstr_scroll_count):
                                down = self.get_element(ver[0], 'auto_id', 'PART_LineDownButton',
                                                        pstr_auto_id='PART_LineDownButton')
                                self.mouse_operations(down, 'click', pstr_click_type='left')
                                self.pint_temp_count += 1
                        elif (self.pstr_scroll_direction.lower() == 'left'):
                            while (self.pint_temp_count < self.pstr_scroll_count):
                                left = self.get_element(horz[0], 'auto_id', 'PART_LineLeftButton',
                                                        pstr_auto_id='PART_LineLeftButton')
                                self.mouse_operations(left, 'click', pstr_click_type='left')
                                self.pint_temp_count += 1
                        elif (self.pstr_scroll_direction.lower() == 'right'):
                            while (self.pint_temp_count < self.pstr_scroll_count):
                                right = self.get_element(horz[0], 'auto_id', 'PART_LineRightButton',
                                                         pstr_auto_id='PART_LineRightButton')
                                self.mouse_operations(right, 'click', pstr_click_type='left')

            else:
                self.__exceptions_thickclient.raise_invalid_parameters(pbool_insert_kibana=True, pbool_trim_log=True,
                                                                       pbool_log_local_exceptions=True,
                                                                       pbool_fail_on_exception=False,

                    if (self.pstr_clickType.lower() is None or "left" in self.pstr_clickType.lower()):

    def window_switch(self, basewindow_title, targetwindow_title):
    """
    Description:
        |  This method should be use to jump / focus of window.

    :param basewindow_title: It is current window title needs
    :type basewindow_title: String
    :param targetwindow_title: it is the title of window ,on which we want to move
    :type targetwindow_title: String
     Examples:
        |  window_switch("Mines","QuickGet"):
        |
    """
    try:
        app = Application(backend="uia").connect(title_re=basewindow_title)
        base_window = app.window(title_re=basewindow_title).set_focus()
        base_window.draw_outline()
        target_window = app.window(title_re=targetwindow_title)
        target_window.draw_outline()
        target_window.set_focus()

    except Exception as e:
        self.__exceptions_generic.raise_generic_exception(
            pstr_message="Error occured while switching windows " + str(e), pbool_insert_kibana=True,
            pbool_trim_log=True, pbool_log_local_exceptions=True, pbool_fail_on_exception=True,
            pbool_use_pytest_format=True)
